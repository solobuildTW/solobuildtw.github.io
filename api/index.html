Quick Start

3 行程式碼開始

# 取得最新台股資訊
curl "https://api.solobuildtw.com/api/v1/stock" \
  -H "X-API-Key: your_key_here" \
  | jq '.data.records[0:5]'
# JavaScript fetch
const BASE = 'https://api.solobuildtw.com';
const r = await fetch(`${BASE}/api/v1/weather`, {
  headers: { 'X-API-Key': process.env.API_KEY }
});
const data = await r.json();
# Python requests
import requests
r = requests.get(
  "https://api.solobuildtw.com/api/v1/exchange",
  headers={"X-API-Key": API_KEY}
)
print(r.json()['data']['rates'])
